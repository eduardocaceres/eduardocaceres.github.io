%% =============================================================================
%% MEMBER DATA VAULT PATTERN - FOCUSED VIEW
%% Demonstrates Hub-Satellite Pattern with Member Entity Only
%% =============================================================================

%% CORE MEMBER HUB - Central Business Entity
    hub_member {
        char_32 member_hk PK "Member Hash Key"
        varchar_50 member_id UK "Business Key - Member ID"
        timestamp load_dts "Load Date Timestamp"
        varchar_100 rec_source "Record Source System"
    }

%% MEMBER DETAIL SATELLITE - Personal Information
    sat_member_details {
        char_32 member_hk PK,FK "References hub_member"
        timestamp load_dts PK "Load Date Timestamp"
        timestamp load_end_dts "Load End Date"
        varchar_100 rec_source "Record Source System"
        char_32 hashdiff "Hash Difference"
        varchar_200 member_name "Full Name"
        date date_of_birth "Date of Birth"
        char_1 gender "Gender Code"
        varchar_500 address "Mailing Address"
        varchar_20 phone_number "Phone Number"
        varchar_100 email "Email Address"
        char_4 ssn_last_4 "Last 4 SSN Digits"
    }

%% MEMBER ELIGIBILITY SATELLITE - Coverage Information
    sat_member_eligibility {
        char_32 member_hk PK,FK "References hub_member"
        timestamp load_dts PK "Load Date Timestamp"
        timestamp load_end_dts "Load End Date"
        varchar_100 rec_source "Record Source System"
        char_32 hashdiff "Hash Difference"
        date effective_date "Coverage Start Date"
        date termination_date "Coverage End Date"
        varchar_50 coverage_type "Coverage Type"
        numeric_10_2 copay_amount "Copay Amount"
        numeric_10_2 deductible_amount "Deductible Amount"
        numeric_10_2 out_of_pocket_max "Max Out of Pocket"
        varchar_20 coverage_status "Active/Inactive Status"
    }

%% PLAN HUB - Related Entity for Context
    hub_plan {
        char_32 plan_hk PK "Plan Hash Key"
        varchar_50 plan_id UK "Business Key - Plan ID"
        timestamp load_dts "Load Date Timestamp"
        varchar_100 rec_source "Record Source System"
    }

%% PLAN SATELLITE - Plan Information
    sat_plan_details {
        char_32 plan_hk PK,FK "References hub_plan"
        timestamp load_dts PK "Load Date Timestamp"
        timestamp load_end_dts "Load End Date"
        varchar_100 rec_source "Record Source System"
        char_32 hashdiff "Hash Difference"
        varchar_200 plan_name "Insurance Plan Name"
        varchar_50 plan_type "Plan Type"
        varchar_200 payer_name "Insurance Company"
        boolean prior_auth_required "Prior Auth Required"
    }

%% MEMBER-PLAN LINK - Enrollment Relationship
    lnk_member_plan {
        char_32 member_plan_hk PK "Link Hash Key"
        char_32 member_hk FK "References hub_member"
        char_32 plan_hk FK "References hub_plan"
        timestamp load_dts "Load Date Timestamp"
        varchar_100 rec_source "Record Source System"
    }
